    <script>
      // Calculator state
      let currentInput = "0"; // Current input string
      let lastOperator = null; // Last operator pressed
      let resetNext = false; // Should reset input on next number

      // Update the calculator display
      function updateDisplay() {
        document.getElementById("display").innerText = currentInput;
      }

      // Append a number to the input
      function appendNumber(num) {
        if (currentInput === "0" || resetNext) {
          currentInput = num;
          resetNext = false;
        } else {
          currentInput += num;
        }
        updateDisplay();
      }

      // Append a decimal point
      function appendDot() {
        // Prevent multiple decimals in the same number
        let parts = currentInput.split(/[\+\-\*\/%]/);
        let last = parts[parts.length - 1];
        if (!last.includes(".")) {
          currentInput += ".";
          updateDisplay();
        }
      }

      // Append an operator (+, -, *, /, %)
      function appendOperator(op) {
        // Prevent two operators in a row
        if (/[+\-*/%]$/.test(currentInput)) {
          currentInput = currentInput.slice(0, -1) + op;
        } else {
          currentInput += op;
        }
        resetNext = false;
        updateDisplay();
      }

      // Calculate the square of the current number
      function square() {
        // Find the last number in the input
        let parts = currentInput.split(/([\+\-\*\/%])/);
        let last = parts.pop();
        if (last === "") last = parts.pop(); // If last is empty, get previous
        let squared = Math.pow(parseFloat(last), 2);
        if (isNaN(squared)) return;
        parts.push(squared);
        currentInput = parts.join("");
        resetNext = true;
        updateDisplay();
      }

      // All Clear (AC) button
      function allClear() {
        currentInput = "0";
        lastOperator = null;
        resetNext = false;
        updateDisplay();
      }

      // Calculate the result
      function calculate() {
        try {
          // Replace '√∑' and '√ó' with JS operators
          let expression = currentInput.replace(/√∑/g, "/").replace(/√ó/g, "*");
          // Evaluate the expression safely
          let result = eval(expression);
          // Handle division by zero and invalid results
          if (result === Infinity || isNaN(result)) {
            currentInput = "Error";
          } else {
            currentInput = result.toString();
          }
          resetNext = true;
        } catch (e) {
          currentInput = "Error";
          resetNext = true;
        }
        updateDisplay();
      }

      // Keyboard support for calculator
      document.addEventListener("keydown", function (event) {
        if (
          document.activeElement.tagName === "BUTTON" ||
          document.activeElement.tagName === "INPUT" ||
          document.activeElement.isContentEditable
        ) {
          return;
        }
        const key = event.key;
        if (key >= "0" && key <= "9") {
          appendNumber(key);
          event.preventDefault();
        } else if (key === ".") {
          appendDot();
          event.preventDefault();
        } else if (["+", "-", "*", "/", "%"].includes(key)) {
          appendOperator(key);
          event.preventDefault();
        } else if (key === "Enter" || key === "=") {
          calculate();
          event.preventDefault();
        } else if (key.toLowerCase() === "c") {
          allClear();
          event.preventDefault();
        } else if (key === "Backspace") {
          if (currentInput.length > 1) {
            currentInput = currentInput.slice(0, -1);
          } else {
            currentInput = "0";
          }
          updateDisplay();
          event.preventDefault();
        }
      });

      // Dark/Light mode toggle
      function toggleMode() {
        document.body.classList.toggle("dark-mode");
        const btn = document.getElementById("modeToggle");
        if (document.body.classList.contains("dark-mode")) {
          btn.innerText = "‚òÄÔ∏è";
        } else {
          btn.innerText = "üåô";
        }
      }
    </script>
